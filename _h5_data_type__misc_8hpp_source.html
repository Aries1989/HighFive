157 inline DataTypeClass convert_type_class(const H5T_class_t& tclass) {
158  switch(tclass) {
159  case H5T_TIME:
160  return DataTypeClass::Time;
161  case H5T_INTEGER:
162  return DataTypeClass::Integer;
163  case H5T_FLOAT:
164  return DataTypeClass::Float;
165  case H5T_STRING:
166  return DataTypeClass::String;
167  case H5T_BITFIELD:
168  return DataTypeClass::BitField;
169  case H5T_OPAQUE:
170  return DataTypeClass::Opaque;
171  case H5T_COMPOUND:
172  return DataTypeClass::Compound;
173  case H5T_REFERENCE:
174  return DataTypeClass::Reference;
175  case H5T_ENUM:
176  return DataTypeClass::Enum;
177  case H5T_VLEN:
178  return DataTypeClass::VarLen;
179  case H5T_ARRAY:
180  return DataTypeClass::Array;
181  case H5T_NO_CLASS:
182  case H5T_NCLASSES:
183  default:
184  return DataTypeClass::Invalid;
185  }
