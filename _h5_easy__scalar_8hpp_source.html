51 // create/write extendible DataSet and write data
52 template <class T, class E = void>
53 struct dump_extend_impl
54 {
55  static DataSet run(File& file,
56  const std::string& path,
57  const T& data,
58  const std::vector<size_t>& idx)
59  {
60  std::vector<size_t> ones(idx.size(), 1);
61 
62  if (file.exist(path)) {
63  DataSet dataset = file.getDataSet(path);
64  std::vector<size_t> dims = dataset.getDimensions();
65  std::vector<size_t> shape = dims;
66  if (dims.size() != idx.size()) {
67  throw detail::error(file, path,
68  "H5Easy::dump: Rank of the index and the existing field do not match");
69  }
70  for (size_t i = 0; i < dims.size(); ++i) {
71  shape[i] = std::max(dims[i], idx[i] + 1);
72  }
73  if (shape != dims) {
74  dataset.resize(shape);
75  }
76  dataset.select(idx, ones).write(data);
77  file.flush();
78  return dataset;
79  }
80 
81  detail::createGroupsToDataSet(file, path);
82  std::vector<size_t> shape = idx;
83  const size_t unlim = DataSpace::UNLIMITED;
84  std::vector<size_t> unlim_shape(idx.size(), unlim);
85  std::vector<hsize_t> chuncks(idx.size(), 10);
86  for (size_t& i : shape) {
87  i++;
88  }
89  DataSpace dataspace = DataSpace(shape, unlim_shape);
90  DataSetCreateProps props;
91  props.add(Chunking(chuncks));
92  DataSet dataset = file.createDataSet(path, dataspace, AtomicType<T>(), props);
93  dataset.select(idx, ones).write(data);
94  file.flush();
95  return dataset;
96  }
97 };
