53 template <typename T>
54 inline void Attribute::read(T& array) const {
55  static_assert(!std::is_const<typename std::remove_reference<T>::type>::value,
56  "read() requires a non-const array to read into");
57  using element_type = typename details::type_of_array<T>::type;
58  const size_t dim_array = details::array_dims<T>::value;
59  DataSpace space = getSpace();
60  DataSpace mem_space = getMemSpace();
61 
62  if (!details::checkDimensions(mem_space, dim_array)) {
63  std::ostringstream ss;
64  ss << "Impossible to read attribute of dimensions "
65  << mem_space.getNumberDimensions() << " into arrays of dimensions "
66  << dim_array;
67  throw DataSpaceException(ss.str());
68  }
69 
70  const DataType mem_datatype = create_and_check_datatype<element_type>();
71 
72  // Apply pre read conversions
73  details::data_converter<T> converter(mem_space);
74 
75  if (H5Aread(getId(), mem_datatype.getId(),
76  static_cast<void*>(converter.transform_read(array))) < 0) {
77  HDF5ErrMapper::ToException<AttributeException>(
78  "Error during HDF5 Read: ");
79  }
80 
81  // re-arrange results
82  converter.process_result(array);
83 }
