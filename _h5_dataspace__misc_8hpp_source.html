203 DataSpace::From(const std::vector<Eigen::Matrix<Value, M, N>>& vec) {
204  using elem_t = Eigen::Matrix<Value, M, N>;
205  std::vector<size_t> dims{
206  std::accumulate(vec.begin(), vec.end(), size_t{0u},

229 inline bool checkDimensions(const DataSpace& mem_space, size_t input_dims) {
230  size_t dataset_dims = mem_space.getNumberDimensions();
231  if (input_dims == dataset_dims)
232  return true;
233 
234  const std::vector<size_t>& dims = mem_space.getDimensions();
235  for (auto i = dims.crbegin(); i != --dims.crend() && *i == 1; ++i)
236  --dataset_dims;
237 
238  if (input_dims == dataset_dims)
239  return true;
240 
241  dataset_dims = dims.size();
242  for (auto i = dims.cbegin(); i != --dims.cend() && *i == 1; ++i)
243  --dataset_dims;
244 
245  if (input_dims == dataset_dims)
246  return true;
247 
248  // The final tests is for scalars
249  return input_dims == 0 && dataset_dims == 1 && dims[dims.size() - 1] == 1;
250 }
