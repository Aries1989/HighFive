195 inline bool checkDimensions(const DataSpace& mem_space, size_t input_dims) {
196  size_t dataset_dims = mem_space.getNumberDimensions();
197  if (input_dims == dataset_dims)
198  return true;
199 
200  const std::vector<size_t> dims = mem_space.getDimensions();
201  for (std::vector<size_t>::const_reverse_iterator i = dims.rbegin();
202  i != --dims.rend() && *i == 1; ++i)
203  --dataset_dims;
204 
205  if (input_dims == dataset_dims)
206  return true;
207 
208  dataset_dims = dims.size();
209  for (std::vector<size_t>::const_iterator i = dims.begin();
210  i != --dims.end() && *i == 1; ++i)
211  --dataset_dims;
212 
213  if (input_dims == dataset_dims)
214  return true;
215 
216  // The final tests is for scalars
217  return input_dims == 0 && dataset_dims == 1 && dims[dims.size() - 1] == 1;
218 }
