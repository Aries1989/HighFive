168 // converter function for hsize_t -> size_t when hsize_t != size_t
169 template<typename Size>
170 inline std::vector<std::size_t> to_vector_size_t(std::vector<Size> vec){
171  static_assert(std::is_same<Size, std::size_t>::value == false, " hsize_t != size_t mandatory here");
172  std::vector<size_t> res(vec.size());
173  std::transform(vec.begin(), vec.end(), res.begin(), [](Size e) { return static_cast<size_t>(e); });
174  return res;
175 }
