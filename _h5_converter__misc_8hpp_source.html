36 inline bool is_1D(const std::vector<size_t>& dims)
37 {

42 inline size_t compute_total_size(const std::vector<size_t>& dims)

44  return std::accumulate(dims.begin(), dims.end(), size_t{1u},

83 // copy single buffer to multi dimensional vector, following dimensions
84 // specified
85 template <typename T>
86 inline typename std::vector<T>::iterator
87 single_buffer_to_vectors(typename std::vector<T>::iterator begin_buffer,
88  typename std::vector<T>::iterator end_buffer,
89  const std::vector<size_t>& dims,
90  const size_t current_dim,
91  std::vector<T>& vec_single_dim) {
92  const auto n_elems = static_cast<long>(dims[current_dim]);
93  const auto end_copy_iter = std::min(begin_buffer + n_elems, end_buffer);
94  vec_single_dim.assign(begin_buffer, end_copy_iter);
95  return end_copy_iter;
96 }

98 template <typename T, typename U = typename type_of_array<T>::type>
99 inline typename std::vector<U>::iterator
100 single_buffer_to_vectors(typename std::vector<U>::iterator begin_buffer,
101  typename std::vector<U>::iterator end_buffer,
102  const std::vector<size_t>& dims,
103  const size_t current_dim,
104  std::vector<std::vector<T>>& vec_multi_dim) {
105  const size_t n_elems = dims[current_dim];
106  vec_multi_dim.resize(n_elems);
107 
108  for (auto& subvec : vec_multi_dim) {
109  begin_buffer = single_buffer_to_vectors(
110  begin_buffer, end_buffer, dims, current_dim + 1, subvec);
111  }
112  return begin_buffer;
113 }

271 struct data_converter<std::vector<T>,

283  transform_write(std::vector<T>& vec) {
284  _vec_align.reserve(compute_total_size(_dims));

356  _c_vec.resize(vec.size() + 1, NULL);

358  return (&_c_vec[0]);

362  (void)vec;
363  vec.resize(_c_vec.size());

365  vec[i] = std::string(_c_vec[i]);
366  }
367 
368  if (_c_vec.empty() == false && _c_vec[0] != NULL) {
369  AtomicType<std::string> str_type;
370  (void)H5Dvlen_reclaim(str_type.getId(), _space.getId(), H5P_DEFAULT,
371  &(_c_vec[0]));
372  }
373  }
