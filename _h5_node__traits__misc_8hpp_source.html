132 template <typename Derivate>
133 inline std::string NodeTraits<Derivate>::getObjectName(size_t index) const {
134  const size_t maxLength = 255;
135  char buffer[maxLength + 1];
136  ssize_t retcode = H5Lget_name_by_idx(
137  static_cast<const Derivate*>(this)->getId(), ".", H5_INDEX_NAME, H5_ITER_INC,
138  index, buffer, static_cast<hsize_t>(maxLength) + 1, H5P_DEFAULT);
139  if (retcode < 0) {
140  HDF5ErrMapper::ToException<GroupException>("Error accessing object name");
141  }
142  const size_t length = static_cast<std::size_t>(retcode);
143  if (length <= maxLength) {
144  return std::string(buffer, length);
145  }
146  std::vector<char> bigBuffer(length + 1, 0);
147  H5Lget_name_by_idx(
148  static_cast<const Derivate*>(this)->getId(), ".", H5_INDEX_NAME, H5_ITER_INC,
149  index, bigBuffer.data(), static_cast<hsize_t>(length) + 1, H5P_DEFAULT);
150  return std::string(bigBuffer.data(), length);
151 }
